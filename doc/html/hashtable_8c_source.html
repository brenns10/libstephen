107 void ht_insert_bucket(smb_ht *table, smb_ht_bckt *pBucket)
108 {
109  smb_ht_bckt *curr;
110  unsigned int index = table->hash(pBucket->key) % table->allocated;
111 
112  if (table->table[index]) {
113  // A linked list already exists here.
114  curr = ht_find_in_bucket(table->table[index], pBucket->key, table->equal);
115  if (table->equal(curr->key, pBucket->key) == 0) {
116  // Key already exists in table
117  curr->value = pBucket->value;
118  ht_bucket_delete(pBucket);
119  } else {
120  // assert curr->next == NULL
121  curr->next = pBucket;
122  table->length++;
123  }
124  } else {
125  // No linked list exists yet
126  table->table[index] = pBucket;
127  table->length++;
128  }

137 void ht_resize(smb_ht *table)
138 {
139  smb_ht_bckt **pOldBuffer;
140  smb_ht_bckt *curr, *temp;
141  int index, oldLength, oldAllocated;
142 
143  // Step one: allocate new space for the table
144  pOldBuffer = table->table;
145  oldLength = table->length;
146  oldAllocated = table->allocated;
147  table->length = 0;
148  table->allocated = ht_next_size(oldAllocated);
149  table->table = smb_new(smb_ht_bckt*, table->allocated);
150 
151  // Zero out the new block too.
152  memset((void*)table->table, 0, table->allocated * sizeof(smb_ht_bckt*));
153 
154  // Step two, add the old items to the new table (no freeing, please)
155  for (index = 0; index < oldAllocated; index++) {
156  if (pOldBuffer[index]) {
157  // a bucket exists here
158  curr = pOldBuffer[index];
159  while (curr) {
160  temp = curr->next; // Hang on to the next pointer...it could be changed
161  // once added to the "new" hash table
162  curr->next = NULL;
163  ht_insert_bucket(table, curr);
164  curr = temp;
165  }
166  }
167  }
168 
169  // Step three: free old data.
170  smb_free(pOldBuffer);
171 }

332 void ht_remove_act(smb_ht *table, DATA key, DATA_ACTION deleter,
333  smb_status *status)
334 {
335  *status = SMB_SUCCESS;
336 
337  smb_ht_bckt *curr, *prev;
338  int index = table->hash(key) % table->allocated;
339 
340  // Stop if the key doesn't exist in the table.
341  if (!table->table[index]) {
342  *status = SMB_NOT_FOUND_ERROR;
343  return;
344  }
345 
346  curr = table->table[index];
347  prev = NULL;
348 
349  while (curr && table->equal(curr->key, key) != 0) {
350  prev = curr;
351  curr = curr->next;
352  }
353 
354  if (curr && prev) {
355  // Found a match, and there is something before it in the linked list.
356  prev->next = curr->next;
357  if (deleter)
358  deleter(curr->value); // Perform user requested action
359  ht_bucket_delete(curr);
360  table->length--;
361  } else if (curr && !prev) {
362  // Found a match, and it is the first in the linked list
363  table->table[index] = curr->next;
364  if (deleter)
365  deleter(curr->value); // Perform user requested action
366  ht_bucket_delete(curr);
367  table->length--;
368  } else {
369  // Didn't find a match in this bucket list
370  *status = SMB_NOT_FOUND_ERROR;
371  }
372 }
