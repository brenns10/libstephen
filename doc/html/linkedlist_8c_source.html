47 void ll_remove_node(smb_ll *list, smb_ll_node *the_node)
48 {

50  smb_ll_node *next = the_node->next;

416 void ll_insert(smb_ll *list, int index, DATA new_data)
417 {
418  if (index <= 0) {
419  ll_prepend(list, new_data);
420  } else if (index >= list->length) {
421  ll_append(list, new_data);
422  } else {
423  smb_ll_node *new_node = ll_create_node(new_data);
424 
425  smb_status status = SMB_SUCCESS;
426  smb_ll_node *current = ll_navigate(list, index, &status);
427 
428  // Since we already checked indices, there can be no errors.
429  assert(status == SMB_SUCCESS);
430 
431  current->prev->next = new_node;
432  new_node->prev = current->prev;
433  new_node->next = current;
434  current->prev = new_node;
435  list->length++;
436  }
437 }

480 int ll_index_of(const smb_ll *list, DATA d, DATA_COMPARE comp)
481 {
482  int idx = 0;
483  smb_ll_node *curr = list->head;
484 
485  while (curr) {
486  if (comp != NULL) {
487  if (comp(curr->data, d) == 0) {
488  return idx;
489  }
490  } else {
491  if (curr->data.data_llint == d.data_llint) {
492  return idx;
493  }
494  }
495  idx++;
496  curr = curr->next;
497  }
498  return -1;
499 }
