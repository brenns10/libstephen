00488 static smb_ll_node *ll_sort_rec(smb_ll_node **head, int length, DATA_COMPARE cmp)
00489 {
00490   smb_ll_node *left = *head;
00491   smb_ll_node *right, *tail, *next;
00492 
00493   // BASE CASE!
00494   if (length <= 1) {
00495     return *head;
00496   }
00497 
00498   // Find midpoint and bisect list.
00499   right = ll_node_navigate(*head, length/2);
00500   right->prev->next = NULL;
00501   right->prev = NULL;
00502 
00503   // Sort each sublist.
00504   ll_sort_rec(&left, length/2, cmp);
00505   ll_sort_rec(&right, length-length/2, cmp);
00506 
00507   // Pick the first element.
00508   if (cmp(left->data, right->data) <= 0) {
00509     *head = left;
00510     tail = left;
00511     left = left->next;
00512   } else {
00513     *head = right;
00514     tail = right;
00515     right = right->next;
00516   }
00517 
00518   // Merge the remaining elements.
00519   while (--length) {
00520     if (left == NULL) {
00521       next = right;
00522       right = right->next;
00523     } else if (right == NULL) {
00524       next = left;
00525       left = left->next;
00526     } else if (cmp(left->data, right->data) <= 0) {
00527       next = left;
00528       left = left->next;
00529     } else {
00530       next = right;
00531       right = right->next;
00532     }
00533     tail->next = next;
00534     next->prev = tail;
00535     next->next = NULL;
00536     tail = next;
00537   }
00538   tail->next = NULL;
00539   return tail;
00540 }
