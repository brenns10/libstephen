147 int find_string(smb_ll *toSearch, char *toFind)
148 {
149  DATA d;
150  char *sCurr;
151  smb_iter iter = ll_get_iter(toSearch);
152  int retVal = -1;
153  smb_status status;
154  while (iter.has_next(&iter)) {
155  d = iter.next(&iter, &status);
156  // We are using 'has_next' to check first!
157  assert(status == SMB_SUCCESS);
158  sCurr = (char *)d.data_ptr;
159  if (strcmp(toFind, sCurr) == 0) {
160  retVal = iter.index-1; // previous index
161  break;
162  }
163  }
164  iter.destroy(&iter);
165  return retVal;
166 }

239 void process_args(smb_ad *data, int argc, char **argv)
240 {
241  char *previous_long_flag = NULL;
242  int previous_flag = EOF;
243 
244  while (argc--) {
245  // At the beginning of the loop, argc refers to number of remaining args,
246  // and argv points at the pointer to the current arg.
247  switch (**argv) {
248  case '-':
249  // Processing new flag type, so set previous variables to invalid.
250  previous_long_flag = NULL;
251  previous_flag = EOF;
252  switch (*(*argv + 1)) {
253  case '-':
254  // Long flag
255  previous_long_flag = process_long_flag(data, *argv);
256  previous_flag = EOF;
257  break;
258  case '\0':
259  // A single '-'...counts as a bare string in my book
260  process_bare_string(data, *argv, previous_long_flag, previous_flag);
261 
262  previous_long_flag = NULL;
263  previous_flag = EOF;
264  break;
265  default:
266  // The input is a short flag
267  previous_flag = process_flag(data, *argv);
268  previous_long_flag = NULL;
269  break;
270  }
271  break;
272 
273  default:
274  // This is a raw string. We first need to check if it belongs to a flag.
275  process_bare_string(data, *argv, previous_long_flag, previous_flag);
276  previous_long_flag = NULL;
277  previous_flag = EOF;
278  break;
279  }
280 
281  argv++;
282  }
283 }
