00031   smb_al visit_queue;
00032   smb_al cap_queue;

00043 
00044   // THIS IS BREADTH FIRST SEARCH

00046

00054     trans_list = al_get(&fs->f->transitions, d.data_llint, &status).data_ptr;
00055     assert(status == SMB_SUCCESS);
00056 
00057     // FOR EACH OUTGOING TRANSITION FROM THE STATE
00058     for (i = 0; i < al_length(trans_list); i++) {
00059

00061       ft = al_get(trans_list, i, &status).data_ptr;
00062       assert(status == SMB_SUCCESS);
00063       d.data_llint = ft->dest;
00064       LDEBUG(" expanding transition to %ld", d.data_llint);

00067       // AND NOT IN THE VISIT QUEUE,
00068       // AND NOT ALREADY IN THE CURRENT STATE LIST...
00069       if (fsm_trans_check(ft, EPSILON) &&

00071           al_index_of(fs->curr, d, &data_compare_int) == -1) {
00072

00074         // CREATE A NEW CAPTURE LIST WITH THE OLD ITEMS

00078         // AND THE CURRENT INDEX, IF THERE IS A CAPTURE IN THIS TRANSITION

00157   // While we have not exhausted our input

00159     SMB_DP("State: %d, Input %Lc\n", state, *input);
00160     // Get the list of transitions from this state

00162     next = -1;

00166       if (fsm_trans_check(trans, *input)) {
00167         if (next == -1) {

00169         } else {
00170           // If there are two matching transitions, error.
00171           fprintf(stderr, "Error: non-deterministic FSM simulated as "
00172                   "deterministic.\n");
00173         }
00174       }
00175     }
00176     // If there are no transitions, fail.
00177     if (next == -1) {
00178       SMB_DP("No available transitions, reject.\n");
00179       return false;
00180     }
00181     state = next;
00182     input++;
00183   }
00184 
00185   SMB_DP("State: %d, Input finished.\n", state);
00186 
00187   // If the state is in the accepting states, we accept, else reject
