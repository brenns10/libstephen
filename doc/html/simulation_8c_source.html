00172   // While we have not exhausted our input
00173   while (*input != '\0') {
00174     SMB_DP("State: %d, Input %Lc\n", state, *input);
00175     // Get the list of transitions from this state
00176     list = (smb_al *) al_get(&f->transitions, state, &status).data_ptr;
00177     next = -1;

00181       if (fsm_trans_check(trans, *input)) {
00182         if (next == -1) {
00183           next = trans->dest;
00184         } else {
00185           // If there are two matching transitions, error.
00186           fprintf(stderr, "Error: non-deterministic FSM simulated as "
00187                   "deterministic.\n");
00188         }
00189       }
00190     }
00191     // If there are no transitions, fail.
00192     if (next == -1) {
00193       SMB_DP("No available transitions, reject.\n");
00194       return false;
00195     }
00196     state = next;
00197     input++;
00198   }
00199 
00200   SMB_DP("State: %d, Input finished.\n", state);
00201 
00202   // If the state is in the accepting states, we accept, else reject
00203   d.data_llint = state;
